 95    async def _wait_for_controller_ready(self, name: str, timeout: int):
 96        """Waits for the controller to be available
 97
 98        Args:
 99            name (str): The name of the controller
100            timeout (int): The time to wait for the controller to be available in seconds
101        """
102        iteration = 0
103        controller = await self._get_controller_instance(name)
104
105        while iteration < timeout:
106            if controller is not None:
107                if (
108                    controller.error_details == "Controller not initialized or disposed"
109                    or controller.error_details == "Initializing controller connection."
110                ):
111                    await self._api_gateway.controller_api.get_current_robot_controller_state(
112                        cell=self._cell_id, controller=controller.host
113                    )
114                elif controller.has_error:
115                    # As long has an error its being initialized
116                    logger.error(controller.error_details)
117                else:
118                    # Controller is ready
119                    return
120
121            logger.info(f"Waiting for {self._cell_id}/{name} controller availability")
122            await asyncio.sleep(1)
123            controller = await self._get_controller_instance(name)
124            iteration += 1
125
126        raise TimeoutError(f"Timeout waiting for {self._cell_id}/{name} controller availability")

128    async def _plan_with_collision_check(
129        self,
130        actions: list[Action],
131        tcp: str,
132        start_joint_position: tuple[float, ...] | None = None,
133        optimizer_setup: wb.models.OptimizerSetup | None = None,
134    ) -> wb.models.JointTrajectory:
135        """
136        This method plans a trajectory and checks for collisions.
137        The collision check only happens if the actions have collision scene data.
138
139        You must provide the exact same collision data into all the actions.
140        Because the underlying API supports collision checks for the whole trajectory only.
141
142        Raises:
143            InconsistentCollisionScenes: If the collision scene data is not consistent across all actions
144
145            Your actions should follow below rules to be considered consistent:
146            1- They all should have the same collision scene data
147            2- They all should have no collision data
148
149            PlanTrajectoryFailed: If the trajectory planning failed including the collision check
150
151        For more information about this API, please refer to the plan_trajectory in the API documentation.
152
153        Args:
154            actions: list of actions to plan, current supported actions are Motion and WriteActions
155                     WriteAction you specify on your path is handled in a performant way.
156                     Please check execute_trajectory.motion_command.set_io for more information.
157            tcp:     The tool to use
158            start_joint_position: The starting joint position, if none provided, current position of the robot is used
159            optimizer_setup: The optimizer setup
160
161        Returns: planned joint trajectory
162
163        """
164        # PREPARE THE REQUEST
165        collision_scenes = self._validate_collision_scenes(actions)
166        start_joint_position = start_joint_position or await self.joints()
167        robot_setup = optimizer_setup or await self._get_optimizer_setup(tcp=tcp)
168
169        motion_commands = CombinedActions(items=tuple(actions)).to_motion_command()  # type: ignore
170
171        static_colliders = None
172        collision_motion_group = None
173        if collision_scenes and len(collision_scenes) > 0:
174            static_colliders = collision_scenes[0].colliders
175
176            motion_group_type = robot_setup.motion_group_type
177            if (
178                collision_scenes[0].motion_groups
179                and motion_group_type in collision_scenes[0].motion_groups
180            ):
181                collision_motion_group = collision_scenes[0].motion_groups[motion_group_type]
182
183        request = wb.models.PlanTrajectoryRequest(
184            robot_setup=robot_setup,
185            start_joint_position=list(start_joint_position),
186            motion_commands=motion_commands,
187            static_colliders=static_colliders,
188            collision_motion_group=collision_motion_group,
189        )
190
191        # EXECUTE THE API CALL
192        plan_trajectory_response = await self._motion_api_client.plan_trajectory(
193            cell=self._cell, plan_trajectory_request=request
194        )
195        if isinstance(
196            plan_trajectory_response.response.actual_instance,
197            wb.models.PlanTrajectoryFailedResponse,
198        ):
199            # TODO: handle partially executable path
200            raise PlanTrajectoryFailed(plan_trajectory_response.response.actual_instance)
201        return plan_trajectory_response.response.actual_instance

203    def _validate_collision_scenes(self, actions: list[Action]) -> list[models.CollisionScene]:
204        motion_count = len([action for action in actions if isinstance(action, Motion)])
205        collision_scenes = [
206            action.collision_scene
207            for action in actions
208            if isinstance(action, CollisionFreeMotion) and action.collision_scene is not None
209        ]
210
211        if len(collision_scenes) != 0 and len(collision_scenes) != motion_count:
212            raise InconsistentCollisionScenes(
213                "Only some of the actions have collision scene. Either specify it for all or none."
214            )
215
216        # If a collision scene is provided, the same should be provided for all the collision scene
217        if len(collision_scenes) > 1:
218            first_scene = collision_scenes[0]
219            if not all(
220                compare_collision_scenes(first_scene, scene) for scene in collision_scenes[1:]
221            ):
222                raise InconsistentCollisionScenes(
223                    "All actions must use the same collision scene but some are different"
224                )
225
226        return collision_scenes

230    async def _plan_collision_free(
231        self,
232        action: CollisionFreeMotion,
233        tcp: str,
234        start_joint_position: list[float],
235        optimizer_setup: wb.models.OptimizerSetup | None = None,
236    ) -> wb.models.JointTrajectory:
237        """
238        This method plans a trajectory and avoids collisions.
239        This means if there is a collision along the way to the target pose or joint positions,
240        It will adjust the trajectory to avoid the collision.
241
242        The collision check only happens if the action have collision scene data.
243
244        For more information about this API, please refer to the plan_collision_free_ptp in the API documentation.
245
246        Args:
247            action: The target pose or joint positions to reach
248            tcp:     The tool to use
249            start_joint_position: The starting joint position, if none provided, current position of the robot is used
250            optimizer_setup: The optimizer setup
251
252        Returns: planned joint trajectory
253
254
255        """
256        target = wb.models.PlanCollisionFreePTPRequestTarget(
257            **action.model_dump(exclude_unset=True)
258        )
259        robot_setup = optimizer_setup or await self._get_optimizer_setup(tcp=tcp)
260
261        static_colliders = None
262        collision_motion_group = None
263        collision_scene = action.collision_scene
264        if collision_scene and collision_scene.colliders:
265            static_colliders = collision_scene.colliders
266
267            if (
268                collision_scene.motion_groups
269                and robot_setup.motion_group_type in collision_scene.motion_groups
270            ):
271                collision_motion_group = collision_scene.motion_groups[
272                    robot_setup.motion_group_type
273                ]
274
275        request: wb.models.PlanCollisionFreePTPRequest = wb.models.PlanCollisionFreePTPRequest(
276            robot_setup=robot_setup,
277            start_joint_position=start_joint_position,
278            target=target,
279            static_colliders=static_colliders,
280            collision_motion_group=collision_motion_group,
281        )
282
283        plan_result = await self._motion_api_client.plan_collision_free_ptp(
284            cell=self._cell, plan_collision_free_ptp_request=request
285        )
286
287        if isinstance(plan_result.response.actual_instance, wb.models.PlanTrajectoryFailedResponse):
288            raise PlanTrajectoryFailed(plan_result.response.actual_instance)
289        return plan_result.response.actual_instance

291    async def _plan(
292        self,
293        actions: list[Action],
294        tcp: str,
295        start_joint_position: tuple[float, ...] | None = None,
296        optimizer_setup: wb.models.OptimizerSetup | None = None,
297    ) -> wb.models.JointTrajectory:
298        if not actions:
299            raise ValueError("No actions provided")
300
301        current_joints = start_joint_position or await self.joints()
302        robot_setup = optimizer_setup or await self._get_optimizer_setup(tcp=tcp)
303
304        all_trajectories = []
305        for batch in split_actions_into_batches(actions):
306            if len(batch) == 0:
307                raise ValueError("Empty batch of actions")
308
309            if isinstance(batch[0], CollisionFreeMotion):
310                motion: CollisionFreeMotion = cast(CollisionFreeMotion, batch[0])
311                trajectory = await self._plan_collision_free(
312                    action=motion,
313                    tcp=tcp,
314                    start_joint_position=list(current_joints),
315                    optimizer_setup=robot_setup,
316                )
317                all_trajectories.append(trajectory)
318                # the last joint position of this trajectory is the starting point for the next one
319                current_joints = tuple(trajectory.joint_positions[-1].joints)
320            else:
321                trajectory = await self._plan_with_collision_check(
322                    actions=batch,
323                    tcp=tcp,
324                    start_joint_position=current_joints,
325                    optimizer_setup=robot_setup,
326                )
327                all_trajectories.append(trajectory)
328                # the last joint position of this trajectory is the starting point for the next one
329                current_joints = tuple(trajectory.joint_positions[-1].joints)
330
331        return combine_trajectories(all_trajectories)

116    ) -> ExecuteTrajectoryRequestStream:

118        yield wb.models.InitializeMovementRequest(trajectory=context.motion_id, initial_location=0)  # type: ignore

121        initialize_movement_response = await anext(response_stream)
122        if isinstance(
123            initialize_movement_response.actual_instance, wb.models.InitializeMovementResponse
124        ):
125            r1 = initialize_movement_response.actual_instance
126            if not r1.init_response.succeeded:
127                raise InitMovementFailed(r1.init_response)
128

131        yield wb.models.StartMovementRequest(

141            # Send the current location to the consume
142            if isinstance(instance, wb.models.Movement):
143                motion_state = movement_to_motion_state(instance)
144                if motion_state:

146

149                if instance.standstill.reason == wb.models.StandstillReason.REASON_MOTION_ENDED:
